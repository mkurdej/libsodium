    PROPERTIES
        PREFIX ""
        PROJECT_LABEL "libsodium"
        INSTALL_RPATH "${LIBSODIUM_LIB_DIR}"
        LINK_FLAGS "${_LINKER_OPTIONS}"
        VERSION "${SODIUM_LIBRARY_VERSION_MAJOR}.${SODIUM_LIBRARY_VERSION_MINOR}"
        SOVERSION "${SODIUM_LIBRARY_VERSION_MAJOR}"
)
